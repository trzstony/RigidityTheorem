\section{Continuity of guessing probability}

\begin{lemma}[Decision-operator form of $p_{\mathrm{guess}}$]
\label{lem:pguess-decision}
\uses{def:pguess}
\lean{RigidityToPguess.pguess_decision}
\notready
For a classical--quantum state $\rho_{XE}$ with $X\in\{0,1\}$,
\[
 p_{\mathrm{guess}}(X\mid E)_\rho = \sup_{0\preceq \Pi\preceq I_{XE}} \Tr(\Pi\rho_{XE}),
\]
where the supremum is over operators of the form $\Pi=|0\rangle\!\langle 0|\otimes M_0 + |1\rangle\!\langle 1|\otimes M_1$ with $\{M_0,M_1\}$ a POVM on $E$.
\end{lemma}

\begin{proof}
Expanding Definition~\ref{def:pguess} gives $\sum_x\Tr(M_x\rho_E^x)=\Tr(\Pi\rho_{XE})$ with the stated $\Pi$. Conversely, any such $\Pi$ corresponds to a valid POVM. Taking the supremum over POVMs is the same as taking the supremum over such $\Pi$.
\end{proof}

\begin{lemma}[Continuity of $p_{\mathrm{guess}}$]
\label{lem:pguess-continuity}
\uses{lem:pguess-decision}
\lean{RigidityToPguess.pguess_continuity}
\notready
For any two classical--quantum states $\rho_{XE}$ and $\sigma_{XE}$,
\[
 p_{\mathrm{guess}}(X\mid E)_\rho \le p_{\mathrm{guess}}(X\mid E)_\sigma + \frac12\|\rho_{XE}-\sigma_{XE}\|_1.
\]
\end{lemma}

\begin{proof}
Set $\Delta:=\rho_{XE}-\sigma_{XE}$. Since $\rho_{XE}$ and $\sigma_{XE}$ are states,
$\Delta$ is Hermitian and $\Tr(\Delta)=0$.

Fix any feasible decision operator $\Pi$. Then
\[
\Tr(\Pi\rho_{XE}) = \Tr(\Pi\sigma_{XE}) + \Tr(\Pi\Delta).
\]

Write the Jordan decomposition $\Delta=\Delta_+-\Delta_-$ with
$\Delta_\pm\succeq 0$ and orthogonal supports. Because $\Tr(\Delta)=0$,
\[
\Tr(\Delta_+)=\Tr(\Delta_-)=\frac12\|\Delta\|_1.
\]
Since $0\preceq \Pi\preceq I$ and $\Delta_-\succeq 0$,
\[
\Tr(\Pi\Delta)=\Tr(\Pi\Delta_+)-\Tr(\Pi\Delta_-)\le \Tr(\Pi\Delta_+)\le \Tr(\Delta_+)=\frac12\|\Delta\|_1.
\]
Therefore
\[
\Tr(\Pi\rho_{XE})\le \Tr(\Pi\sigma_{XE})+\frac12\|\rho_{XE}-\sigma_{XE}\|_1.
\]
Now take the supremum over feasible $\Pi$ and use Lemma~\ref{lem:pguess-decision}.
\end{proof}

\begin{lemma}[Guessing probability for a uniform and independent bit]
\label{lem:pguess-uniform}
\uses{def:pguess}
\lean{RigidityToPguess.pguess_uniform}
\notready
Let $\tau_E$ be any state. If
\[
\sigma_{XE}=\frac{I_X}{2}\otimes \tau_E,
\]
then $p_{\mathrm{guess}}(X\mid E)_\sigma=\tfrac12$.
\end{lemma}

\begin{proof}
For any POVM $\{M_0,M_1\}$, one has
\[
\sum_x \Tr(M_x\sigma_E^x)=\frac12\Tr(M_0\tau_E)+\frac12\Tr(M_1\tau_E)=\frac12\Tr((M_0+M_1)\tau_E)=\frac12.
\]
Taking the supremum gives $p_{\mathrm{guess}}=\tfrac12$.
\end{proof}
