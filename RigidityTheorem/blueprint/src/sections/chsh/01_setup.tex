\section{CHSH Setup}

\begin{definition}[Binary observable]
\label{def:binary-observable}
\lean{Approximate_Rigidity.IsBinaryObservable}
A binary observable on a complex Hilbert space $H$ is a linear map $A : H \to H$ that is self-adjoint and satisfies $A^2 = \id$.
\end{definition}

\begin{definition}[CHSH strategy]
\label{def:chsh-strategy}
\lean{Approximate_Rigidity.CHSHStrategy}
A CHSH strategy consists of:
\[
S = (\psi, A_0, A_1, B_0, B_1),
\]
where $\psi \in H_A \otimes H_B$ is a unit vector and each $A_i$, $B_j$ is a binary observable on the corresponding local space.
\end{definition}

\begin{definition}[CHSH operator and bias]
\label{def:chsh-bias}
\uses{def:chsh-strategy}
\lean{Approximate_Rigidity.CHSH_op, Approximate_Rigidity.chshBias}
Given a strategy $S$, the associated CHSH operator is
\[
\CHSH(A_0,A_1,B_0,B_1)
= A_0 \otimes B_0 + A_0 \otimes B_1 + A_1 \otimes B_0 - A_1 \otimes B_1.
\]
Its bias on the state $\psi$ is
\[
\Bias(S) = \Re \ip{\psi}{\CHSH(A_0,A_1,B_0,B_1)\psi}.
\]
\end{definition}

\begin{definition}[Error function]
\label{def:delta}
\lean{Approximate_Rigidity.delta, Approximate_Rigidity.cConst}
The main spectral bookkeeping uses
\[
\delta(\eps) := \eps + 4\sqrt{\cconst \eps},
\qquad
\cconst = 128 \sqrt{2}.
\]
\end{definition}

\begin{remark}
The final theorems in the Lean development are phrased in terms of the bias lower bound
\[
\Bias(S) \ge 2\sqrt{2} - \eps.
\]
This is the normalization that interacts directly with the Bell-space operator $K$ in the later sections.
\end{remark}
