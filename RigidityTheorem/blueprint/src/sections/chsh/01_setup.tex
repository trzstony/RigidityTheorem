\section{Setup and basic definitions}

\begin{definition}[Systems and reduced states]
\label{def:systems}
\lean{RigidityToPguess.Systems}
\notready
Let $A$ and $B$ be qubit systems and let $E$ be an arbitrary quantum system.
A tripartite state is a density operator $\rho_{ABE}$ on $A\otimes B\otimes E$.
We write $\rho_{AB}:=\Tr_E(\rho_{ABE})$ for the reduced state on $AB$.
\end{definition}

\begin{definition}[Bell state]
\label{def:bell-phi-plus}
\lean{RigidityToPguess.PhiPlus}
\notready
The Bell state $|\Phi^+\rangle\in A\otimes B$ is
\[
|\Phi^+\rangle := \frac{|00\rangle + |11\rangle}{\sqrt{2}},\qquad
\Phi := |\Phi^+\rangle\!\langle\Phi^+|.
\]
\end{definition}

\begin{definition}[Canonical CHSH operator on two qubits]
\label{def:chsh-operator}
\uses{def:systems}
\lean{RigidityToPguess.CHSH}
\notready
Let $X,Z$ denote Pauli operators on a qubit.  Define observables
\[
A_0:=Z,\quad A_1:=X,\quad
B_0:=\frac{Z+X}{\sqrt 2},\quad B_1:=\frac{Z-X}{\sqrt 2}.
\]
The associated CHSH operator on $A\otimes B$ is
\[
\mathrm{CHSH} := A_0\otimes B_0 + A_0\otimes B_1 + A_1\otimes B_0 - A_1\otimes B_1.
\]
\end{definition}

\begin{definition}[Classical--quantum state after Alice's key measurement]
\label{def:rho-xe}
\uses{def:systems}
\lean{RigidityToPguess.rhoXE}
\notready
Let $\{|0\rangle,|1\rangle\}$ be the computational basis of $A$, and set $P_x:=|x\rangle\!\langle x|_A$.
From $\rho_{ABE}$ define subnormalised conditional states
\[
\rho_E^x := \Tr_{AB}\big((P_x\otimes I_{BE})\,\rho_{ABE}\,(P_x\otimes I_{BE})\big),\qquad x\in\{0,1\}.
\]
The induced classical--quantum state is
\[
\rho_{XE} := \sum_{x\in\{0,1\}} |x\rangle\!\langle x|_X\otimes \rho_E^x.
\]
\end{definition}

\begin{definition}[Guessing probability]
\label{def:pguess}
\uses{def:rho-xe}
\lean{RigidityToPguess.pGuess}
\notready
For the classical--quantum state $\rho_{XE}$, the guessing probability is
\[
p_{\mathrm{guess}}(X\mid E)_{\rho}
:=\sup_{\{M_0,M_1\}} \sum_{x\in\{0,1\}} \Tr(M_x\,\rho_E^x),
\]
where the supremum ranges over POVMs on $E$, i.e. $M_x\succeq 0$ and $M_0+M_1=I_E$.
\end{definition}
